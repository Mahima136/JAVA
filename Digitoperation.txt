package digitoperation;
import java.util.Scanner;

public class DigitOperation {
public static void main(String[] args) {
	Scanner sc = new Scanner(System.in);
	System.out.print("Enter a number: ");
	int num = sc.nextInt();
	int originalNum = num;
	int count = 0,sum = 0,product = 1;
	int evenCount = 0;
	String evenDigits = "";
	int reverse = 0;
	while (num > 0) {
		int digit = num % 10;
		count++;
		sum += digit;
		product *= digit;
		if (digit % 2 == 0) {
			evenCount++;
			evenDigits = digit + " " + evenDigits;
		}
		reverse = reverse * 10 + digit;
		num /= 10;
	}
	System.out.println("\nOriginal Number: " + originalNum);
	System.out.println("Count of Digits: " + count);
	System.out.println("Sum of Digits: " + sum);
	System.out.println("Product of Digits: " + product);
	System.out.println("Count of Even Digits: " + evenCount);
	System.out.println("Even Digits: " + evenDigits);
	System.out.println("Reversed Number: " + reverse);
	
			
		}
	
}
